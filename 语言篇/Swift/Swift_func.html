Swift 函数

1、函数命名规则

/// 有返回值
func eat(food:String) -> String {
    return "中午吃\(food)"
}
eat(food: "米饭")

/// 无返回值、参数为可选型
func eat(food:String?)  {
    print("中午吃\(food ?? "空气")")
}
eat(food: nil)

2、使用元组作为函数返回值

func findMaxAndMin(numbsers : [Int]) -> (max:Int, min:Int) {

    guard numbsers.count > 0 else {
        return (max : 0, min : 0)
    }

    var minValue = numbsers[0]
    var maxValue = numbsers[0]

    for value in numbsers {
        minValue = minValue < value ? minValue : value
        maxValue = maxValue > value ? maxValue : value
    }
    return (maxValue, minValue)
}

var arr : [Int] = [1, 2, 3, 4, 5]
findMaxAndMin(numbsers: arr)

使用元组作为返回值，并将返回值设置为可选型！

func request(info:String?) -> (message:String, code:Int)? {
    guard let info = info else {
        return nil
    }
    return ("失败",404)
}
request(info: "get")

3、内部参数名、外部参数名、函数命名规则

Swift 设计的函数名，更像自然语言！默认使用 OC 的命名规则，参数带有参数名：

  • 外部参数名：外部调用者使用的参数名，但也可以在函数体内部使用的参数名；
  • 内部参数名：区分于外部参数名，仅用于函数体内部使用；
/// 参数 time、food 属于外部参数名
func eat(time:String, food:String) -> String {
    return "\(time)的食物是\(food)"
}
eat(time: "早上", food: "牛奶")

函数名中的参数，带有介词使得函数更具有可读性，但是当介词出现在函数内部时，会出现语义上的歧义！

func eat(time:String, withFood:String) -> String {
    return "\(time)的食物是\(withFood)" /// 内部参数的语义问题
}

eat(time: "早上", withFood: "牛奶")

此时可以使用内部参数名、外部参数名将场景区分

/// 外部参数名 withFood
/// 内部参数名 food
func eat(time:String, withFood food:String) -> String {
    return "\(time)的食物是\(food)" /// 函数体内部使用 内部参数名
}

/// 调用者使用外部参数名
eat(time: "早上", withFood: "牛奶")

在某些场景中，调用者使用外部参数名可能会显得语义冗余，如下述代码：

func add(num1:Int, num2:Int) -> Int {
    return num1 + num2
}

/// add 操作后面，明显是两个整数！此时 参数名 num1、num2 如果不显示，可能效果更好
add(num1: 10, num2: 20)

在外部参数名没有起到提示作用，反而冗余时，可以将外部参数名隐藏掉：

/// 在外部参数名的位置，使用 _ 替代
func add(_ num1:Int, _ num2:Int) -> Int {
    return num1 + num2
}

/// 调用者无需关注参数名，直接传入两个数值即可
add(10, 20) /// 此时的语义，已经很清晰了

4、默认参数值

C++ 的函数支持为参数设置默认值：

  • 有默认值的参数必须在参数列表的最右端；
  • 无实参则用默认值，否则实参覆盖默认值；

同 C++ 一样，Swift 也可以为函数参数设置默认值

  • Swift 的默认值参数不必放在参数列表最右端；
  • 但建议按照 C++ 的要求，也将默认值参数放在最右边；
/// 设置参数默认值：不必设置在最右端
func add(num1: Int, num2 : Int = 10, num3:Int) {
    print("\(num1) + \(num2)")
}

add(num1: 100, num3: 1)
add(num1: 19, num2: 21, num3: 20)

add(num1: 100, num3: 1) /// num2 使用默认值
add(num1: 19, num2: 21, num3: 20)

5、变长参数值

对于一个函数，最多只能有一个变长参数！如果有多个变长参数，就无法确定哪个参数属于哪个部分！

func sum( numbers : Float ...) {
    var sum: Float = 0
    for va in numbers { /// 将变长参数当做一个数组对待
        sum += va
    }
    print(sum)
}

sum(numbers: 1,2,3,4,5,6)

6、函数的常量参数、变量参数、inout 参数

在前文讲述的函数参数，仅仅是读取其属性值，并没有改变参数值！ 当我们尝试修改传入的参数值时，编译器会提示常量参数不可修改的错误 Cannot assign to value: 'message' is a 'let' constant ! 因为 Swift 默认传入的参数是一个常量参数，设置为常量，对编译器的性能有一定的优化效果！ 如果想要修改并使用参数值，需要显式的声明参数为变量类型:

但这种声明为变量的方法，仅仅作用于函数内部，函数外部的调用者并没有被改变！

在 Swift 中需要使用 inout 关键字修改传入的参数值！

func swap_test(_ a : inout Int, _ b : inout Int) {
    (a, b) = (b, a) /// 通过元组的方式来交换其值
//    let c = a
//    a = b
//    b = c
}

var a = 10, b = 20
swap_test(&a, &b) /// 类似于 C++ 的引用方式

7、函数型变量

函数也可以作为一个变量来使用：函数型变量！

func add(_ a:Int, _ b:Int) -> Int {
    return a + b
}

/// 将函数当作变量，赋值给一个常量
let otherAdd = add /// 常量类型 (Int, Int) -> Int
let otherAdd : (Int, Int) -> Int = add /// 显式声明其类型
otherAdd(10, 20)

函数型变量使用的场景很多，比如当作一个变量传递到另一个函数参数中！

var strArr = ["2", "6", "9", "1"]

/// 自定义比较规则
func sortString(_ a:String, _ b:String) -> Bool {
    return a > b
}

/// 函数当作变量，传递到参数中
strArr.sorted(by: sortString)

8、函数式编程

如下代码所示，针对一批小数值，编写一个函数来增大：

func handleData1(_ datas: inout [Int]) {
    for (index, value) in datas.enumerated() {
        datas[index] = value + 10
    }
}
var data1 = [50, 52, 53, 49, 59]
handleData1(&data1)

现在又来了一批大数据，再次编写一个函数来缩减：

func handleData2(_ datas: inout [Int]) {
    for (index, value) in datas.enumerated() {
        datas[index] = value / 10
    }
}
var data2 = [150, 252, 353, 149, 259]
handleData2(&data2)

分析上述两段函数，虽然具体逻辑不同，但大致模版确实类似的！假如每次碰到一批数据就编写一个函数，那么垃圾冗余代码将由此而生！

可以将上述代码再次抽象：

/// 将共同逻辑抽象为一个函数
func handleData(_ datas: inout [Int],by rule:(Int) -> Int) {
    for (index, value) in datas.enumerated() {
        datas[index] = rule(value)
    }
}

/// 将差异性编成一个函数，作为变量传递到抽象函数的参数中
func handleDataRule1(_ value: Int) -> Int {
    return value + 10
}
var data1 = [50, 52, 53, 49, 59]
handleData(&data1, by: handleDataRule1)

/// 定义修改规则即可
func handleDataRule2(_ value: Int) -> Int {
    return value / 10
}
var data2 = [150, 252, 353, 149, 259]
handleData(&data2, by: handleDataRule2)

8.1、map 函数

map 函数将数组中的每个元素扫描一次，然后根据某种自定义规则做一次修改，再将修改后的元素存储回去！

func handleDataRule1(_ value: Int) -> Int {
    return value + 10
}
var data1 = [50, 52, 53, 49, 59]
data1.map(handleDataRule1)

8.2、filter 函数

filter 函数将数组中的每个元素扫描一次，然后根据某种自定义规则做一次比较

  • 符合需求的返回 true ，保留该元素
  • 不符合需求的返回 false，剔除该元素
  • 最后返回一个新的数组
/// 保留小于 200 的数值
func filterRule(_ value: Int) -> Bool {
    return value < 200
}
var data2 = [150, 252, 353, 149, 259]
data2.filter(filterRule)

8.3、reduce 函数

reduce 函数又称聚合函数，将数组中的每个元素扫描一次，然后根据某种自定义规则做一次聚合

  • 自定义的规则函数需要传入两个值，返回一个值
  • reduce 函数也需要两个参数：一个是初值，一个是聚合规则：
/// 将数组中的所有元素相加，求和
func reduceRule(_ value1: Int, _ value2 : Int) -> Int {
    return value1 + value2
}
var data2 = [150, 252, 353, 149, 259]
/// 数组的和，初值为 0
data2.reduce(0, reduceRule)

针对一些特殊的符号，Swift 已经封住成函数：

var data2 = [150, 252, 353, 149, 259]
/// Swift 的求和函数
data2.reduce(0, +) /// Swift 的运算符重载

9、返回函数类型和函数嵌套

上级规定的售票规则：

/// 学生收半价
func chargingRules_student(_ price : Float) -> Float {
    return price / 2.0
}

/// 成人旺季收高价票
func chargingRules_adult(_ price : Float) -> Float {
    return price * 1.1
}

/// 老人收三折票价
func chargingRules_old(_ price : Float) -> Float {
    return price * 0.3
}

景区根据不同年龄，返回不同的收费规则：

/// 返回一个函数型变量
func scenicCharging(_ age: Int) -> (Float) -> Float {
    if age > 60 {
        return chargingRules_old
    }
    if age < 18 {
        return chargingRules_student
    }
    return chargingRules_adult
}

收票员在窗口，根据游客的年龄大小收取不同的费用：

func ticket(_ age: Int, _ price : Float) ->  Float {
    let rule = scenicCharging(age) /// 拿到具体的收费规则，一个函数型变量
    return rule(price)  /// 调用该函数，去计价
}

ticket(30, 10)

在上述代码中，景区收费规则作为一个函数单独存在，仅仅是语义上的明晰，又可以不必单独存在，此时可以嵌套到售票员函数

func ticket(_ age: Int, _ price : Float) ->  Float {   

    /// 将一个函数嵌套到另一个函数中
    /// 满足清晰的语义，又不必单独存在
    func scenicCharging(_ age: Int) -> (Float) -> Float {
        if age > 60 {
            return chargingRules_old
        }
        if age < 18 {
            return chargingRules_student
        }
        return chargingRules_adult
    }

    let rule = scenicCharging(age)
    return rule(price)
}
